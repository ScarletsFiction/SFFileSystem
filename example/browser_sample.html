// This script will be executed on global scope
var myFile = new SFFileSystem({
	home:'myRootFolder',
	initError:function(){
		console.error("Can't initialize filesystem");
	},
	callback:function(){
		console.log("Initialized");
	}
});

// Example: Create and read file in a folder
myFile.home.newFolder('myFolder', function(myFolder){
	// Callback after the folder was created

	myFolder.newFile('myText', function(myText){
		// Callback after the file was created

		myText.write('hello world!', function(){
			// Callback after the write was finished

			myText.read(console.log);
		});
	});
});
